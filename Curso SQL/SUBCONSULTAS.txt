SUB CONSULTAS (FROM)

SELECT [LISTA DE CAMPOS]
FROM [LISTA DE TABLAS], (	SELECT [LISTA DE CAMPOS]
			FROM [TABLA/S]) [ALIAS_SUBCONSULTA]
[WHERE]
[GROUP BY]
[HAVING]
[ORDER BY]


-- La subconsulta en el FROM devuelve una tabla.

Ejercicio 1  (CLAUSULA FROM)

SELECT D.NOMBRE, SALARIO_PROMEDIO_DEPTO.SRPOM, E.NOMBRE, E.SULEDO
FROM 	(SELECT codigo_departamento, avg(sueldo) AS 'sprom'
	FROM empleado
	GROUP BY codigo_departamento) SALARIO_PROMEDIO_DEPTO, 
	EMPLEADO E,
	DEPARTAMENTO D
WHERE SALARIO_PROMEDIO_DEPTO.CODIGO_DEPARTAMENTO = E.CODIGO_DEPARTAMENTO 
AND DEPARTAMENTO.CODIGO_DEPARTAMENTO = E.CODIGO_DEPARTAMENTO
AND EMPLEADO.SALARIO > SALARIO_PROMEDIO_DEPTO.SPROM
